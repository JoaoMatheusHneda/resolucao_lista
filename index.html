    set.seed(123)

    posicao_valores_na <<- which(is.na(z))
    valores_na <<- z[posicao_valores_na]

vizinhos_mais_proximos <- function(z, x, funcao_distancia, num_vizinhos_mais_proximos){
    vizinhos_mais_proximos_z_i <<- list()

    for(i in 1:length(z)){
        vizinhos_mais_proximos_z_i[[i]] <<- x[order(funcao_distancia(z[i], x))[1:num_vizinhos_mais_proximos]]
        names(vizinhos_mais_proximos_z_i)[[i]] <<- paste0("z_",i," = ",z[i])
    }
    
    return(vizinhos_mais_proximos_z_i)
}
